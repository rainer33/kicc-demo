-- MariaDB DDL for kicc_demo
-- Use DB
CREATE DATABASE IF NOT EXISTS `kicc_demo` CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE `kicc_demo`;

-- Drop child tables first
DROP TABLE IF EXISTS `payment_refunds`;
DROP TABLE IF EXISTS `payments`;
DROP TABLE IF EXISTS `orders`;
DROP TABLE IF EXISTS `idempotency_records`;
DROP TABLE IF EXISTS `audit_logs`;
DROP TABLE IF EXISTS `kiosk_sessions`;
DROP TABLE IF EXISTS `test_histories`;

CREATE TABLE `orders` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `order_id` VARCHAR(64) NOT NULL,
  `order_name` VARCHAR(120) NOT NULL,
  `buyer_name` VARCHAR(60) NOT NULL,
  `amount` DECIMAL(15,2) NOT NULL,
  `status` VARCHAR(32) NOT NULL,
  `created_at` DATETIME(6) NOT NULL,
  `updated_at` DATETIME(6) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_orders_order_id` (`order_id`),
  KEY `idx_orders_updated_at` (`updated_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `payments` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `order_id` VARCHAR(64) NOT NULL,
  `amount` DECIMAL(15,2) NOT NULL,
  `order_name` VARCHAR(120) NOT NULL,
  `buyer_name` VARCHAR(60) NOT NULL,
  `status` VARCHAR(20) NOT NULL,
  `pg_transaction_id` VARCHAR(120) NULL,
  `fail_reason` VARCHAR(255) NULL,
  `refunded_amount` DECIMAL(15,2) NOT NULL DEFAULT 0.00,
  `created_at` DATETIME(6) NOT NULL,
  `updated_at` DATETIME(6) NOT NULL,
  `approved_at` DATETIME(6) NULL,
  `canceled_at` DATETIME(6) NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_payments_order_id` (`order_id`),
  KEY `idx_payments_status_updated` (`status`, `updated_at`),
  KEY `idx_payments_updated_at` (`updated_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `payment_refunds` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `payment_id` BIGINT NOT NULL,
  `amount` DECIMAL(15,2) NOT NULL,
  `refund_transaction_id` VARCHAR(120) NOT NULL,
  `reason` VARCHAR(255) NULL,
  `created_at` DATETIME(6) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_refunds_payment_created` (`payment_id`, `created_at`),
  CONSTRAINT `fk_payment_refunds_payment`
    FOREIGN KEY (`payment_id`) REFERENCES `payments` (`id`)
    ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `idempotency_records` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `idem_key` VARCHAR(120) NOT NULL,
  `order_id` VARCHAR(64) NULL,
  `request_hash` VARCHAR(64) NOT NULL,
  `created_at` DATETIME(6) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_idempotency_idem_key` (`idem_key`),
  KEY `idx_idempotency_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `audit_logs` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `action` VARCHAR(60) NOT NULL,
  `target_id` VARCHAR(80) NOT NULL,
  `detail` VARCHAR(600) NOT NULL,
  `created_at` DATETIME(6) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_audit_target_created` (`target_id`, `created_at`),
  KEY `idx_audit_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `kiosk_sessions` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `session_id` VARCHAR(64) NOT NULL,
  `kiosk_id` VARCHAR(40) NOT NULL,
  `order_name` VARCHAR(120) NOT NULL,
  `amount` DECIMAL(15,2) NOT NULL,
  `status` VARCHAR(20) NOT NULL,
  `payment_method` VARCHAR(20) NULL,
  `order_id` VARCHAR(64) NULL,
  `terminal_transaction_id` VARCHAR(80) NULL,
  `last_message` VARCHAR(600) NULL,
  `created_at` DATETIME(6) NOT NULL,
  `updated_at` DATETIME(6) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_kiosk_sessions_session_id` (`session_id`),
  KEY `idx_kiosk_sessions_updated_at` (`updated_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `test_histories` (
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `run_id` VARCHAR(64) NOT NULL,
  `scenario_name` VARCHAR(160) NOT NULL,
  `category` VARCHAR(40) NOT NULL,
  `status` VARCHAR(20) NOT NULL,
  `environment` VARCHAR(80) NOT NULL,
  `api_endpoint` VARCHAR(160) NULL,
  `http_method` VARCHAR(12) NULL,
  `idempotency_key` VARCHAR(120) NULL,
  `concurrency_level` INT NULL,
  `network_condition` VARCHAR(80) NULL,
  `expected_result` VARCHAR(1000) NOT NULL,
  `actual_result` VARCHAR(2000) NOT NULL,
  `request_payload` TEXT NULL,
  `response_payload` TEXT NULL,
  `notes` VARCHAR(1200) NULL,
  `executed_by` VARCHAR(80) NOT NULL,
  `started_at` DATETIME(6) NOT NULL,
  `finished_at` DATETIME(6) NULL,
  `duration_ms` BIGINT NULL,
  `created_at` DATETIME(6) NOT NULL,
  `updated_at` DATETIME(6) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_test_histories_run_id` (`run_id`),
  KEY `idx_test_histories_started_at` (`started_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
